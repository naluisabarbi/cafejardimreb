    <script>
        document.addEventListener('DOMContentLoaded', function () {
            const tabButtons = document.querySelectorAll('.tab-btn');
            const tabContents = document.querySelectorAll('.tab-content');

            tabButtons.forEach(button => {
                button.addEventListener('click', () => {
                    const tabId = button.dataset.tab;

                    tabButtons.forEach(btn => btn.classList.remove('active'));
                    button.classList.add('active');

                    tabContents.forEach(content => {
                        if (content.id === tabId) {
                            content.classList.remove('hidden');
                        } else {
                            content.classList.add('hidden');
                        }
                    });
                });
            });
            
            const navLinks = document.querySelectorAll('.nav-link');
            const sections = document.querySelectorAll('main section');

            const observerOptions = {
                root: null,
                rootMargin: '0px',
                threshold: 0.4
            };

            const observer = new IntersectionObserver((entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        navLinks.forEach(link => {
                            link.classList.toggle('active', link.getAttribute('href').substring(1) === entry.target.id);
                        });
                    }
                });
            }, observerOptions);

            sections.forEach(section => {
                observer.observe(section);
            });

            const cityData = [
                { city: 'São Paulo', stores: 60, color: '#6B4F4B' },
                { city: 'Rio de Janeiro', stores: 45, color: '#A4B494' },
                { city: 'Belo Horizonte', stores: 25, color: '#6B4F4B' },
                { city: 'Brasília', stores: 7, color: '#A4B494' },
                { city: 'Salvador', stores: 15, color: '#6B4F4B' },
                { city: 'Curitiba', stores: 10, color: '#A4B494' }
            ];

            const ctx = document.getElementById('expansionChart').getContext('2d');
            const expansionChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: cityData.map(d => d.city),
                    datasets: [{
                        label: 'Lojas Planejadas (Projeção)',
                        data: cityData.map(d => d.stores),
                        backgroundColor: cityData.map(d => d.color),
                        borderColor: cityData.map(d => d.color),
                        borderWidth: 1
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        y: {
                            beginAtZero: true,
                            title: {
                                display: true,
                                text: 'Número de Lojas'
                            }
                        }
                    },
                    plugins: {
                        legend: {
                            display: false
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let label = context.dataset.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed.y !== null) {
                                       label += context.parsed.y;
                                    }
                                    return label;
                                }
                            }
                        },
                        title: {
                            display: true,
                            text: 'Projeção de Lojas em Grandes Capitais (2030)'
                        }
                    }
                }
            });
        });
    </script>
